\large{\textbf{Module ferme – contamination du lait}}
\normalsize
\begin{itemize}
	\item \tcbox{Approche directe} — \textit{Listeria} et \textit{E. coli} - \myred{avec données de terrain}
    $$
    Y_{\rm Lm}^{\rm lait} \sim \mypurple{\lognormal}
    \text{ \mybrown{\citep{sanaa04}}}
    $$
    $$
    Y_{\rm Ecoli}^{\rm lait} \sim \mypurple{\lognormal} 
    \text{ \mybrown{\citep{basak24mra}}}
    $$
    \item \tcbox{Approche indirecte} — STEC et \textit{Salmonella}
    \begin{align*}
        Y_{\rm pathogène}^{\rm lait} =
        Y_{\rm Ecoli}^{\rm lait}
        \cdot
        \left( 
        Y^{\rm fèces}_{\rm pathogène} /
        Y^{\rm fèces}_{\rm Ecoli} 
        \right)
    \end{align*}

    \begin{itemize}
        \item \myred{Pourquoi ?} Limite de détection :  concentration faible
        \item \myred{Comment ?} Utilisant les concentrations 
        $Y^{\rm fèces}_{\rm pathogène}$ et $Y_{\rm Ecoli}^{\rm lait}$
    \end{itemize}
    \item \myblue{Sorties :} concentrations simulées $Y_{x}^{\rm lait}$ de pathogènes($x$) dans le lait
\end{itemize}

\large{\textbf{Module consumer – estimation du risque}}
\normalsize
\begin{itemize}
    \item \myblue{Dose dans une portion de fromage (25 g)} 
	    $\Gamma_x =$ \myred{nombre$_x$} $\times$ \myred{taille$_x$}

    \item \myblue{Probabilité de maladie pour 25 g de fromage}
	    \begin{itemize}
		    \item Fonction (epidemiologique) dose-réponse : $\myred{P_{x}}(\text{âge}, \Gamma_{x})$
            \end{itemize}
    \item \myred{Risque de maladie} moyen pour \myblue{un lot particulier} de fromages 
    \begin{align*}
	R^{\rm lot}_{x}(\mygreen{\xi_x}) = \sum_{\mypurple{\text{âge}}}
	g({\mypurple{\text{âge}}}) \cdot
        \int_{\Gamma_{x}}\myred{P_{x}}(\mypurple{\text{âge}},
	    \Gamma_x) \text{d}\gamma_x
    \end{align*}
    \scriptsize
    \item[]$\rightarrow g(\mypurple{\text{âge}})$ : proportion de consommation de fromage par groupe d’âge 
    \item[]$\rightarrow \mygreen{\xi_x}$ : variables aléatoires décrivant un lot
    \normalsize
    \item \myred{Risque moyen de maladie} (indépendant du lot) : $R_x = \int_{\mygreen{\xi_x}} R^{\rm lot}_{x}(\mygreen{\xi_x}) \text{d}\mygreen{\xi_x}$
\end{itemize}

\large{\textbf{Comment choisir les scénarios optimaux ?}}
\normalsize
\begin{itemize}
	\item Nous utilisons le concept \myred{d'optimalité de Pareto}
	\item L'idée est d'obtenir un \myblue{compromis efficace entre les 
		différents objectifs} \myred{plutôt qu'une solution unique}
	\item Il cherche des solutions où aucune amélioration n'est 
		possible dans un objectif \myred{sans détériorer un autre}
	\item Les solutions Pareto-optimales sont \myblue{non dominées}, 
		offrant un compromis efficace entre les objectifs
	\item Elles forment \myred{le front de Pareto}, et les décideurs 
		choisissent la meilleure solution selon leurs préférences
\end{itemize}

\large{\textbf{Estimation du front de Pareto : \myred{difficultés}}}
\normalsize
\begin{itemize}
        \item \myblue{Coûteux} : \myred{3 jours de calcul} pour évaluer les $1500$ scénarios
        \item \myblue{Stochastique} : chaque scénario est évalué avec \myred{une incertitude}
        \item Proposer un algorithme \myblue{d'optimisation bayésienne}
        \begin{itemize}
                \item Évaluation \myred{séquentielle} des scénarios « importants »
		\item \myred{Régression par processus gaussien} pour tous les scénarios
		\item Comparaison avec des \myred{rectangles d’incertitude} \mybrown{\citep{barracosa21}}
        \end{itemize}
\end{itemize}
\begin{itemize}
	\item Avec l'algorithme : \myblue{4 h de calcul} avec une précision de
                \myred{$96\%$}
        \begin{itemize}
                \item Ma thèse : \mybrown{\citet{basak24these}}
        \end{itemize}
\end{itemize}
